<script>
    const apiKey = ""; 
    const messageList = document.getElementById('message-list');
    const chatInput = document.getElementById('chat-input');
    const chatMessages = document.getElementById('chat-messages');
    const loading = document.getElementById('loading');
    const skipText = document.getElementById('skip-text');
    const escLabel = document.getElementById('esc-ui-label');
    const typingIndicator = document.getElementById('typing-indicator');
    const reportModal = document.getElementById('report-modal');

    const personalities = [
        "RUDE: Impatient and mean. Persona: 'ugh', 'boring', 'next'.",
        "HORNY_BOT: A fake bot trying to get you to a link. Skips after a few messages.",
        "WHOLESOME: Very nice, stays for long chats.",
        "WEIRD: Creepy and random. Mentions strange smells or basement living.",
        "BORED: One word replies like 'k' or 'lol'. Skips very quickly.",
        "ELITIST: Snobby. 'I only talk to people with high IQs.'"
    ];

    let currentPersonality = "";
    let chatHistory = [];
    let isConnected = false;

    async function fetchStrangerReply(userMessage) {
        if (!isConnected) return;
        typingIndicator.style.display = "block";
        chatMessages.scrollTop = chatMessages.scrollHeight;

        const systemPrompt = `You are a stranger on Omegle. YOUR PERSONALITY: ${currentPersonality}. 
        You have the power to DISCONNECT. If you want to skip, start message with [SKIP]. 
        Otherwise, reply in short, lowercase internet slang.`;

        chatHistory.push({ role: "user", parts: [{ text: userMessage }] });

        try {
            const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
                method: "POST",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify({ 
                    contents: chatHistory.slice(-10), 
                    systemInstruction: { parts: [{ text: systemPrompt }] } 
                })
            });
            
            const result = await response.json();
            let replyText = result.candidates?.[0]?.content?.parts?.[0]?.text || "k";
            
            setTimeout(() => {
                typingIndicator.style.display = "none";
                
                if (replyText.includes("[SKIP]")) {
                    const finalMsg = replyText.replace("[SKIP]", "").trim();
                    if (finalMsg) appendMsg("Stranger", finalMsg, "stranger-msg");
                    strangerDisconnect();
                } else {
                    chatHistory.push({ role: "model", parts: [{ text: replyText }] });
                    appendMsg("Stranger", replyText, "stranger-msg");
                }
            }, 800 + Math.random() * 1500);

        } catch (error) {
            typingIndicator.style.display = "none";
        }
    }

    function strangerDisconnect() {
        isConnected = false;
        const div = document.createElement('div');
        div.className = "disconnected-msg";
        div.innerText = "Stranger has disconnected.";
        messageList.appendChild(div);
        
        skipText.innerText = "NEW";
        escLabel.innerText = "ESC";
        chatMessages.scrollTop = chatMessages.scrollHeight;
    }

    function handleSend() {
        const val = chatInput.value.trim();
        if(!val || !isConnected) return;
        appendMsg("You", val, "you-msg");
        chatInput.value = "";
        fetchStrangerReply(val);
    }

    function appendMsg(sender, text, cls) {
        const div = document.createElement('div');
        div.className = "mb-2 text-sm";
        div.innerHTML = `<span class="${cls}">${sender}:</span> ${text}`;
        messageList.appendChild(div);
        chatMessages.scrollTop = chatMessages.scrollHeight;
    }

    function handleSkip() {
        if (!isConnected) {
            resetChat();
            return;
        }

        if (skipText.innerText === "STOP") {
            skipText.innerText = "REALLY?";
        } else {
            resetChat();
        }
    }

    function resetChat() {
        isConnected = false;
        messageList.innerHTML = "";
        chatHistory = [];
        currentPersonality = personalities[Math.floor(Math.random() * personalities.length)];
        
        skipText.innerText = "STOP";
        escLabel.innerText = "ESC";
        loading.style.display = "flex";
        
        const status = document.createElement('p');
        status.className = "status-msg";
        status.innerText = "Connecting to server...";
        messageList.appendChild(status);

        setTimeout(() => {
            loading.style.display = "none";
            isConnected = true;
            status.innerText = "You're now chatting with a random stranger. Say hi!";
        }, 1200);
    }

    function openModal() { reportModal.style.display = "flex"; }
    function closeModal() { reportModal.style.display = "none"; }
    function submitReport(r) {
        alert("Thank you. We have received your report for: " + r + ". Searching for a new stranger...");
        closeModal();
        resetChat();
    }

    chatInput.addEventListener('keydown', (e) => {
        if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); handleSend(); }
        if (e.key === 'Escape') handleSkip();
    });

    window.onload = resetChat;
</script>
